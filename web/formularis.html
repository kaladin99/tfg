                            function validarFormulari() {
                            const camps = [
                            { id: 'nom', missatge: 'nom-error' },
                            { id: 'cognom', missatge: 'cognom-error' },
                            { id: 'email', missatge: 'email-error' }

                            const error = document.getElementById(camp.missatge);

                            });

Camp de contrasenya amb validació dinàmica

Mínim 8 caràcters amb almenys una xifra

Codi del camp de contrasenya 

<label for="pwd">Contrasenya:</label>
<input type="password" id="pwd" name="pwd" aria-describedby="pwd-help pwd-msg" minlength="8" required>
<small id="pwd-help">Mínim 8 caràcters amb almenys una xifra</small>
<p id="pwd-msg" role="status" aria-live="polite"></p>

<script>
(function () {
  const pwd = document.getElementById('pwd');
  const msg = document.getElementById('pwd-msg');
  const reNum = /\d/;
  function update() {
    const v = pwd.value || '';
    let text = ''; let invalid = false;
    if (v.length && v.length < 8) { text = 'La contrasenya és massa curta'; invalid = true; }
    else if (v.length >= 8 && !reNum.test(v)) { text = 'Afig almenys una xifra'; invalid = true; }
    else if (v.length >= 8 && reNum.test(v)) { text = 'Contrasenya amb requisits mínims complits'; }
    msg.textContent = text;
    if (invalid) pwd.setAttribute('aria-invalid','true');
    else pwd.removeAttribute('aria-invalid');
  }
  pwd.addEventListener('input', update);
})();
</script>
